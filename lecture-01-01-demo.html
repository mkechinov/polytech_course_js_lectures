Проверка введенных данных в форме

Проверяет, что в поле введено число от 1 до 100. Если это не так, то выводит сообщение об ошибке.

var value = document.getElementById('validate').value;
value = parseInt(value);
if(value < 1) {
	document.getElementById('validate-result').innerText = 'Слишком маленькое число';
} else if (value > 100) {
	document.getElementById('validate-result').innerText = 'Слишком большое число';
} else if (isNaN(value)) {
	document.getElementById('validate-result').innerText = 'Это не число';
} else {
	document.getElementById('validate-result').innerText = 'Хорошее число';
}
